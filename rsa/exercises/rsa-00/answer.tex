\textbf{Given:}
\begin{itemize}
    \item Ciphertext $c = 230539333248$
    \item Public key $(N, e) = (100000016300000148701, 7)$
\end{itemize}

\textbf{Step 1:} Factor $N$ into primes.

Observing that $N$ is close to $10^{20}$, we try factors near $10^{10}$:
\begin{align}
    p &= 10000000019 \\
    q &= 9999999991
\end{align}

Verification: $p \times q = 10000000019 \times 9999999991 = 100000016300000148701 = N$ \checkmark

\textbf{Step 2:} Compute $\phi(N) = (p-1)(q-1)$.
\begin{align}
    \phi(N) &= 10000000018 \times 9999999990 \\
    &= 100000016280000148702
\end{align}

\textbf{Step 3:} Find $d$ such that $ed \equiv 1 \pmod{\phi(N)}$.
\begin{align}
    d &= e^{-1} \pmod{\phi(N)} \\
    &= 7^{-1} \pmod{100000016280000148702} \\
    &= 85714299668571555517
\end{align}

\textbf{Step 4:} Decrypt the ciphertext.
\begin{align}
    m &= c^d \pmod{N} \\
    &= 230539333248^{85714299668571555517} \pmod{100000016300000148701} \\
    &= 112358132134
\end{align}

\textbf{Answer:} 
\begin{itemize}
    \item Plaintext: $112358132134$
    \item Private key: $(N, d) = (100000016300000148701, 85714299668571555517)$
\end{itemize}
